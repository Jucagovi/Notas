create view listado_evaluaciones
with
  (security_invoker = on) as
select
  evalua.id_discente,
  evalua.peso,
  evalua.nota,
  e.nombre as "nombre_evaluacion",
  e.id_evaluacion,
  e.fecha_ini,
  e.fecha_fin,
  d.nombre as "nombre_discente",
  d.apellidos,
  p.id_practica,
  p.numero,
  p.enunciado,
  p.unidad,
  p.nombre as "nombre_practica"
from
  "evaluan" as evalua,
  "Evaluaciones" as "e",
  "Discentes" as "d",
  "Practicas" as "p"
where
  evalua.id_evaluacion = e.id_evaluacion
  and evalua.id_discente = d.id_discente
  and evalua.id_practica = p.id_practica;
